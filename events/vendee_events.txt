	fire_only_once = yes

    picture = GFX_report_event_parliament
    immediate = { }

    option = {
        name = vendee.301.a   # Approve subsidies and advisors
        ai_chance = { factor = 65 }
        VND = {
            add_timed_idea = { idea = VND_british_subsidies_idea days = 180 }

        name = vendee.301.b   # Decline

        VND = { country_event = { id = vendee.314 days = 1 } }
        add_opinion_modifier = { target = VND modifier = embargo }
    }
}
